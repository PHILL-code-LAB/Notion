class DoublyLinkedList{

    DoublyLinkedList(){
      head = nullptr;
      tail = nullptr;

  void add(string data ){
    Node* newNode= new Node{data, nullptr,nullptr};
    if(head == nullptr){
    //當沒有node 存在於 head 與 tail 之間
    head = newNode;
    tail = newNode;
    now = newNode;
    }else{
		//有 node 存在
    tail->next = newNode; // newNode加到尾巴
    newNode->prev =tail; // newNode 本身元素, 也要把tail加入雙向鏈結
    tail =newNode; //把尾巴指向新的tail
    now = newNode;
  }
  }

    if(now==nullptr){
      return;
    }
    if(now==tail){
      now=tail;
      return;
    }
    now=now->next;
  }
  void backward(){
    if(now==nullptr){
      return;
    }
    if(now==head){
      now=head;
      return;
    }
    now=now-> prev;
  }

// Stack - array (陣列版) 

#include <iostream>
using namespace std;
#define MAX_SIZE 20 //宣告陣列的容量  為一個定值  -- Step1 

class StackArray{ // Step2 
  private: 
    int arr[MAX_SIZE];
    int top ; //指向最上面的元素 以便pop 出來 ,內容為index 
  public:
    StackArray():top(-1){}  //初始化  //Step3 
    
    void push(int data){ //把data push 進 StackArray
      if(top>MAX_SIZE-1){  // MAZ_SIZE - 1 => 0-19  //Step 5  
        cout << "OverFlow" <<endl;
        return;
      }
      arr[++top] =data; // 先遞增(從 index 0 開始存)  先遞增top 再存入 (Step 4) 
    }
    
    int  pop(void){ // 按照順序彈出 不需要給值  //Step6 
      if(top<0){   // Step 8 
        cout << "UnderFlow" <<endl;
        return -9999; //return 一個錯誤值
      }
      return arr[top--];  //<- 克服掉 return 後 無法處理的問題  // Step 7  先使用在遞減
      
      
    }
    bool isEmpty(void){
      return (top==-1)? true:false;
    }
    
    void print(){
      cout <<"[";
      for(int i=0;i<=top;i++){
        cout << arr[i] << "|";
      }
      cout <<endl;
    }
};

int main() 
{
  
    StackArray abc;
    abc.push(10);
    abc.push(20);
    abc.push(30);
    abc.print();
    cout << "pop-> " << abc.pop()<<endl;
    cout << "pop-> " << abc.pop()<<endl;
    abc.print();
    return 0;
}
